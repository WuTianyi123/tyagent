# tyagent 子代理系统 v2 实现计划

> **For Hermes:** 不考虑迁移复杂度，只求最优设计。对标 Codex v2 子代理框架。

**Goal:** 将 tyagent 子代理系统从 one-shot `chat()` 模型升级为常驻 thread + 层级 task_name + 全双工 mailbox 通信模型，完全对齐 Codex v2。

**Architecture:** 子代理从一次性 `asyncio.Task` 升级为常驻 `_agent_loop()`，各自拥有独立 mailbox；通信通过 `Mailbox` 替代 `EventCollector`；寻址从 UUID 改为层级 task_path（/root/task_a/sub_b）；新增 `send_message`/`followup_task`/`resume_agent` 工具；消息身份标记改为 `role: assistant` + `author` + `phase: commentary`。

**Tech Stack:** Python 3.11+, asyncio, 无需新依赖。

# tyagent/subagent/task_tree.py
"""Hierarchical task-path registry for sub-agent lifecycle management."""

from __future__ import annotations

import re
from typing import Any, Callable, Dict, List, Optional


class TaskNode:
    """A node in the task tree representing one spawned agent."""
    
    __slots__ = ("name", "parent", "children", "agent", "status")
    
    def __init__(self, name: str, agent: Any = None):
        self.name = name
        self.parent: Optional[TaskNode] = None
        self.children: Dict[str, TaskNode] = {}
        self.agent = agent
        self.status = "running"

    @property
    def path(self) -> str:
        """Canonical task path like /root/main/subtask."""
        if self.parent is None:
            return f"/{self.name}"
        return f"{self.parent.path}/{self.name}"

    def descendant_paths(self) -> List[str]:
        """All paths in the subtree rooted at this node."""
        paths = [self.path]
        for child in self.children.values():
            paths.extend(child.descendant_paths())
        return paths


class TaskTree:
    """Hierarchical task-path registry for a session.
    
    The root node is created once per TyAgent.  Each spawn_task creates
    a child node whose path is the concatenation of the parent path and
    the user-supplied task_name (after sanitisation).
    
    Paths can be referenced by:
    - Relative name (e.g. "db_query") → resolved in the caller's subtree
    - Absolute path (e.g. "/root/task_a/db_query")
    """
    
    _VALID_NAME = re.compile(r"^[a-z][a-z0-9_]*$")
    
    def __init__(self, root_name: str = "root"):
        self._root = TaskNode(root_name)
        self._by_path: Dict[str, TaskNode] = {self._root.path: self._root}
    
    @property
    def root_path(self) -> str:
        return self._root.path
    
    def sanitize_name(self, name: str) -> str:
        """Normalise a user-supplied task_name to a path-safe identifier."""
        cleaned = re.sub(r"[^a-z0-9_]", "_", name.lower())
        cleaned = re.sub(r"_+", "_", cleaned).strip("_")
        return cleaned or "task"
    
    def resolve(self, parent_path: str, task_name: str) -> str:
        """Resolve a relative or absolute task_name to a canonical path.
        
        Absolute paths (starting with /) are used as-is.
        Relative names are joined to parent_path.
        """
        if task_name.startswith("/"):
            return task_name
        name = self.sanitize_name(task_name)
        return f"{parent_path}/{name}"
    
    def register(self, parent_path: str, task_name: str, agent: Any) -> str:
        """Register a new agent in the tree. Returns the canonical path."""
        path = self.resolve(parent_path, task_name)
        if path in self._by_path:
            raise ValueError(f"Task path already exists: {path}")
        
        parts = path.strip("/").split("/")
        node = self._root
        for i, part in enumerate(parts[1:], start=1):  # skip root
            if part in node.children:
                node = node.children[part]
            else:
                new_node = TaskNode(part, agent if i == len(parts) - 1 else None)
                new_node.parent = node
                node.children[part] = new_node
                node = new_node
        
        self._by_path[path] = node
        return path
    
    def lookup(self, path: str) -> Optional[TaskNode]:
        """Find a node by canonical path."""
        return self._by_path.get(path)
    
    def resolve_relative(self, caller_path: str, target: str) -> Optional[str]:
        """Resolve a relative or absolute target to a canonical path."""
        if target.startswith("/"):
            return target if target in self._by_path else None
        
        # Try relative to caller path
        candidate = f"{caller_path}/{self.sanitize_name(target)}"
        if candidate in self._by_path:
            return candidate
        
        # Try as a name in the caller's sibling scope
        if caller_path != "/root":
            parent = caller_path.rsplit("/", 1)[0]
            candidate = f"{parent}/{self.sanitize_name(target)}"
            if candidate in self._by_path:
                return candidate
        
        return None
    
    def descendants(self, path: str) -> List[str]:
        """Return paths of all nodes in the subtree."""
        node = self._by_path.get(path)
        if node is None:
            return []
        return node.descendant_paths()
    
    def unregister(self, path: str) -> Optional[TaskNode]:
        """Remove a node and all its descendants from the tree."""
        node = self._by_path.pop(path, None)
        if node is None:
            return None
        for desc_path in node.descendant_paths():
            self._by_path.pop(desc_path, None)
        if node.parent:
            node.parent.children.pop(node.name, None)
        return node
    
    def all_paths(self) -> List[str]:
        return list(self._by_path.keys())
    
    def filter_by_prefix(self, prefix: str) -> List[str]:
        return [p for p in self._by_path if p.startswith(prefix)]
```

# tyagent/subagent/mailbox.py
"""Mailbox for inter-agent communication (Codex v2 model).

Each agent thread has one Mailbox.  Messages are enqueued by sibling
agents and dequeued by the owning agent's event loop.

Two message kinds:
  - InterAgentMessage:  mid-task progress or collaboration
  - FinalNotification:  child completed, includes summary
"""

from __future__ import annotations

import asyncio
import time
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional


@dataclass
class InterAgentMessage:
    """A message sent from one agent to another (mid-task or final)."""
    author: str           # canonical task_path of sender
    recipient: str        # canonical task_path of receiver
    content: str          # message body
    trigger_turn: bool = False  # whether to start a new LLM turn on delivery
    timestamp: float = field(default_factory=time.time)


@dataclass  
class FinalNotification:
    """Final result from a completed child agent."""
    task_path: str
    success: bool
    summary: Optional[str]
    error: Optional[str]
    duration_seconds: float
    timestamp: float = field(default_factory=time.time)


# Union type for mailbox items
MailboxItem = InterAgentMessage | FinalNotification


class Mailbox:
    """Per-agent mailbox for inter-agent communication.
    
    Each agent (main or sub) has exactly one Mailbox.  Messages arrive
    via send() from other agents; the owner drains them via drain().
    
    Uses an asyncio.Event for efficient wait_next() polling.
    """
    
    def __init__(self, owner_path: str = "/root"):
        self.owner_path = owner_path
        self._queue: List[MailboxItem] = []
        self._event = asyncio.Event()
    
    def send(self, item: MailboxItem) -> None:
        """Enqueue a message. Thread-safe — may be called from any context."""
        self._queue.append(item)
        self._event.set()
    
    def drain(self) -> List[MailboxItem]:
        """Dequeue all pending items and clear the event."""
        items = self._queue[:]
        self._queue.clear()
        self._event.clear()
        return items
    
    def peek(self) -> bool:
        return bool(self._queue)
    
    async def wait_next(self, timeout: Optional[float] = None) -> bool:
        """Block until a message arrives (or timeout)."""
        if self._queue:
            return True
        try:
            await asyncio.wait_for(self._event.wait(), timeout=timeout)
            return True
        except asyncio.TimeoutError:
            return False
```
